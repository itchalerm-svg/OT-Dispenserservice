// ลิงก์ Web App URL จาก Google Apps Script (ให้พี่เอาลิงก์ exec ของพี่มาวางตรงนี้) const GOOGLE_WEB_APP_URL = "https://script.google.com/macros/s/AKfycbymtxgwQRCwEHHT61DI2Qwk30gClhi7o-AoT7i-I2H1JSyvKQfbrbVomX3fQXUo8Ie_9w/exec"; // ฟังก์ชันดึงข้อมูลและประมวลผลสำหรับทำฟิลเตอร์สลับช่วงเวลาที่ฝั่งหน้าจอ (Client-side) async function getFullDashboardData() { try { // 1. ยิงดึงข้อมูลดิบจากท่อ Google Web App API const response = await fetch(GOOGLE_WEB_APP_URL); if (!response.ok) throw new Error("เครือข่ายตอบกลับไม่ถูกต้อง"); const result = await response.json(); // ตรวจสอบโครงสร้างข้อมูลที่ส่งมาจากแอปสคริปต์ if (!result || !result.records) { console.error("ไม่พบรูปแบบ records ในข้อมูลที่ส่งมา"); return { records: [] }; } const records = result.records; console.log("ดึงข้อมูลสำเร็จ จำนวนรายการ:", records.length); // 2. ส่งข้อมูลออกไปให้เพื่อนเอาไปใช้วาดกราฟ หรือทำตารางคัดกรองช่วงเวลาต่อได้เลย return { records: records // ในนี้จะมี timestamp, dateString, dateKey, dateLabel, name, company, hours, detail ครบถ้วนตามหลังบ้าน }; } catch (err) { console.error("เกิดข้อผิดพลาดในการดึงข้อมูลแดชบอร์ด:", err); return { records: [], error: err.toString() }; } }
